      const date = document.getElementById('dateInput').value;
      const reason = document.getElementById('specialReason').value;

        return;
      }

      const duration = 5 * 1000;
      const end = Date.now() + duration;

        confetti({
          particleCount: 5,
          angle: 60,
          spread: 55,
          origin: { x: 0 },
        });
        confetti({
          particleCount: 5,
          angle: 120,
          spread: 55,
          origin: { x: 1 },
        });

        if (Date.now() < end) {
          requestAnimationFrame(frame);
        }
      })();
    });

function openLetter(letterType) {
  const letterTitle = document.getElementById("letterTitle");
  const letterText = document.getElementById("letterText");
  const modal = document.getElementById("letterModal");

  // 📝 Customize your letter content here
  const letters = {
    happy: {
      title: "Open When You're Happy 😊",
      text: "Your smile is my favorite thing in the whole world. When you're happy, I'm the happiest person alive. Keep glowing, love! 💖"
    },
    sad: {
      title: "Open When You're Sad 😢",
      text: "I wish I could wrap you in the biggest hug right now. It's okay to feel down sometimes, but remember — I'm always here for you. You're never alone. 🥺💕"
    },
    angry: {
      title: "Open When You're Angry 😡",
      text: "Take a deep breath, baby. I know things can be frustrating, but you're strong and kind and thoughtful. I'm proud of you — even in hard moments. ❤️"
    },
    lonely: {
      title: "Open When You're Lonely 🥺",
      text: "Even from miles away, I’m sending love and cuddles. You're never really alone, because my heart is always with you. 🌙✨"
    },
    loved: {
      title: "Open When You Feel Loved 💕",
      text: "Because you ARE loved — deeply, completely, endlessly. Thank you for being in my life. I love you more than words can ever say. 💌"
    }
  };

  // ✨ Load the correct letter content
  if (letters[letterType]) {
    letterTitle.textContent = letters[letterType].title;
    letterText.textContent = letters[letterType].text;
    modal.classList.remove("hidden");
  }
}

function closeLetter() {
  document.getElementById("letterModal").classList.add("hidden");
}
